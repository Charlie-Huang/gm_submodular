num_points=50
class St(gm_submodular.DataElement):
    x=np.random.rand(num_points,2)
    dist_v=dist.pdist(x)
    Y=np.ones(num_points)
    def getCosts(self):
        return np.ones(num_points)
    def getDistances(self):
        d=dist.squareform(self.dist_v)
        return np.multiply(d,d)
S=St()
